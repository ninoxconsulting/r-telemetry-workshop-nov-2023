Create an Area of Interest (AOI)

scott <- read_sf("clean_data/caribou.shp") |> 
  filter(commnts == "Scott")

scott_aoi <- st_bbox(scott) |> 
  st_as_sfc()
st_write(scott_aoi, "clean_data/scott_aoi.gpkg")
